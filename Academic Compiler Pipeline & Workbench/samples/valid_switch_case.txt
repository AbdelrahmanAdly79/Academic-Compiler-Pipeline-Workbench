#include <iostream>
using namespace std;

int main() {
    int x = 2 + (3 * 4);
    int y = 0;

    cout << "x = " << x << endl;

    switch (x) {
    case 1:
        y = x + 1;
        break;
    case 2:
        y = y * 2;
        break;
    default:
        y = 0;
    }

    cout << "y = " << y << endl;
    return 0;
}
